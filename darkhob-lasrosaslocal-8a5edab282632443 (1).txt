# backend/api/auth_backend.py
from rest_framework.authentication import BaseAuthentication
from rest_framework import exceptions
from django.apps import apps

from .jwt_utils import decode_jwt  # 👈 importante: NO importamos auth_views aquí


class UsuarioJWTAuthentication(BaseAuthentication):
    """
    Autenticación personalizada que:
    - Lee 'Authorization: Bearer <token>'
    - Decodifica el token con decode_jwt
    - Busca al Usuario (modelo de 'api') usando el 'sub' del payload
    - Coloca ese Usuario en request.user
    """

    def authenticate(self, request):
        # Puedes usar request.headers o get_authorization_header, las dos valen.
        auth_header = request.headers.get('Authorization', '')

        if not auth_header.startswith('Bearer '):
            # No hay token -> DRF prueba otras auth classes (si las hay) o deja user como Anonymous
            return None

        token = auth_header.split(' ', 1)[1].strip()
        if not token:
            raise exceptions.AuthenticationFailed('Token vacío')

        # Decodificar el JWT
        try:
            payload = decode_jwt(token)
        except Exception:
            raise exceptions.AuthenticationFailed('Token inválido o expirado')

        user_id = payload.get('sub')
        if not user_id:
            raise exceptions.AuthenticationFailed('Token sin identificador de usuario')

        Usuario = apps.get_model('api', 'Usuario')
        try:
            usuario = Usuario.objects.select_related('rol').get(usuario_id=int(user_id))
        except Usuario.DoesNotExist:
            raise exceptions.AuthenticationFailed('Usuario no encontrado')

        # 🔴 Parche importante: DRF espera que el user tenga .is_authenticated
        # Lo añadimos dinámicamente sin tocar el modelo ni la base de datos.
        setattr(usuario, 'is_authenticated', True)

        # DRF espera un tuple (user, auth)
        return (usuario, None)

def get_usuario_model():
    """Importación perezosa para evitar problemas de orden de carga."""
    return apps.get_model('api', 'Usuario')


class LoginView(APIView):
    # Endpoint público: no requiere autenticación previa
    authentication_classes = []
    permission_classes = []

    def post(self, request):
        Usuario = get_usuario_model()
        username = (request.data.get('username') or '').strip()
        password = request.data.get('password') or ''
        if not username or not password:
            return Response({'detail': 'Usuario y contraseña son obligatorios.'}, status=400)

        try:
            user = Usuario.objects.select_related('rol').get(nombre_usuario=username)
        except Usuario.DoesNotExist:
            return Response({'detail': 'Usuario o contraseña incorrectos.'}, status=401)

        if not check_pwd(password, user.hash_contrasena):
            return Response({'detail': 'Usuario o contraseña incorrectos.'}, status=401)

        user_data = UsuarioSerializer(user).data
        access = create_jwt({
            'sub': user.usuario_id,          
            'username': user.nombre_usuario,
            'role': user.rol.nombre_rol,
        }, minutes=60)

        return Response({'access': access, 'user': user_data}, status=200)

def _key_fingerprint():
    """Huella de la SECRET_KEY para confirmar que es estable sin exponerla."""
    return hashlib.sha256((settings.SECRET_KEY or "").encode()).hexdigest()[:12]


class DiagKeyView(APIView):
    authentication_classes = []
    permission_classes = []

    def get(self, request):
        return Response({"key_fingerprint": _key_fingerprint()}, status=200)


class DiagTokenView(APIView):
    authentication_classes = []
    permission_classes = []

    def get(self, request):
        auth = request.headers.get('Authorization', '')
        token = auth.split(' ', 1)[1] if auth.startswith('Bearer ') else ''
        if not token:
            return Response({"error": "no token"}, status=400)

        # payload sin verificar (para inspeccionar)
        try:
            noverify = jwt.decode(
                token,
                options={"verify_signature": False, "verify_exp": False, "verify_iat": False},
                algorithms=["HS256"]
            )
        except Exception as e:
            return Response({"stage": "decode_noverify", "error": type(e).__name__, "msg": str(e)}, status=400)

        # validación real con SECRET_KEY del proceso (reusamos decode_jwt del módulo)
        try:
            verified = decode_jwt(token)
            return Response({
                "key_fingerprint": _key_fingerprint(),
                "verified": True,
                "payload": verified,
                "noverify": noverify,
            }, status=200)
        except Exception as e:
            return Response({
                "key_fingerprint": _key_fingerprint(),
                "verified": False,
                "error_type": type(e).__name__,
                "error_msg": str(e),
                "noverify": noverify,
            }, status=401)

# backend/api/jwt_utils.py
import datetime
import jwt
from django.conf import settings


def create_jwt(payload: dict, minutes=60):
    """
    Crea un JWT con exp/iat. Fuerza 'sub' a string (PyJWT v2 lo requiere).
    """
    exp = datetime.datetime.utcnow() + datetime.timedelta(minutes=minutes)
    to_encode = {**payload, 'exp': exp, 'iat': datetime.datetime.utcnow()}
    if 'sub' in to_encode:
        to_encode['sub'] = str(to_encode['sub'])
    return jwt.encode(to_encode, settings.SECRET_KEY, algorithm='HS256')


def decode_jwt(token: str):
    """
    Valida el JWT con leve tolerancia de reloj.
    """
    return jwt.decode(
        token,
        settings.SECRET_KEY,
        algorithms=['HS256'],
        options={"require": ["exp", "iat"]},
        leeway=10,  # ±10s por si el reloj está levemente desfasado
    )

# backend/api/security.py
from django.contrib.auth.hashers import check_password, make_password
def make_pwd(raw: str) -> str: return make_password(raw)
def check_pwd(raw: str, hashed: str) -> bool:
    if hashed and hashed.startswith('pbkdf2_'):
        return check_password(raw, hashed)
    return raw == hashed

class HistorialInventarioSerializer(serializers.ModelSerializer):
    # --- Campos de LECTURA ---
    nombre_articulo = serializers.CharField(
        source='inventario.nombre_articulo',
        read_only=True
    )

    # Lo calculamos a mano para ser flexibles
    nombre_usuario = serializers.SerializerMethodField()
    fecha = serializers.DateTimeField(read_only=True)

    # ID amigable para el front
    id = serializers.IntegerField(source='pk', read_only=True)

    # --- Campo de ESCRITURA ---
    inventario = serializers.PrimaryKeyRelatedField(
        queryset=Inventario.objects.all()
    )

    class Meta:
        model = Historialinventario
        fields = [
            'id',
            'fecha',
            'tipo_movimiento',
            'cantidad',
            'motivo',
            'inventario',
            'nombre_articulo',
            'nombre_usuario',
        ]
        extra_kwargs = {
            'motivo': {'required': True, 'allow_blank': False},
            'tipo_movimiento': {'required': True, 'allow_blank': False},
        }

    def create(self, validated_data):
        """
        Asigna fecha actual y mapea request.user (Usuario)
        al modelo que el FK espera (Usuarios).
        """
        request = self.context.get('request')

        # Si tenemos request y el usuario es del modelo Usuario (login)
        if request and hasattr(request, 'user') and isinstance(request.user, Usuario):
            u_login = request.user

            try:
                # Mapeo Usuario -> Usuarios usando usuario_id (comparten ID)
                usuario_hist = Usuarios.objects.get(usuario_id=u_login.usuario_id)
                validated_data['usuario'] = usuario_hist
            except Usuarios.DoesNotExist:
                # Si no hay coincidencia, dejamos usuario vacío (se verá como "Sistema")
                pass

        # Fecha actual
        validated_data['fecha'] = timezone.now()
        return super().create(validated_data)

    def get_nombre_usuario(self, obj):
        """
        Devuelve el nombre del usuario que hizo el movimiento.
        Si no hay usuario, devolvemos 'Sistema'.
        """
        u = getattr(obj, 'usuario', None)
        if not u:
            return 'Sistema'

        # Usuarios tiene nombre_completo y nombre_usuario
        return (
            getattr(u, 'nombre_completo', None)
            or getattr(u, 'nombre_usuario', None)
            or str(u)
        )

    # Campo para recibir la contraseña en texto plano

    class Meta:
        model = Usuario
        fields = ['nombre_completo', 'nombre_usuario', 'password', 'rol_id']

    def create(self, validated_data):
        # 1. Extraer contraseña y rol_id
        raw_password = validated_data.pop('password')
        rol_id = validated_data.pop('rol_id')

        # 2. Buscar la instancia del Rol
        from .models import Rol
        try:
            rol_instance = Rol.objects.get(pk=rol_id)
        except Rol.DoesNotExist:
            raise serializers.ValidationError({"rol_id": "El rol especificado no existe."})

        # 3. Encriptar contraseña
        hashed_password = make_pwd(raw_password)

        # 4. Crear el usuario manualmente
        usuario = Usuario.objects.create(
            nombre_completo=validated_data['nombre_completo'],
            nombre_usuario=validated_data['nombre_usuario'],
            hash_contrasena=hashed_password,
            rol=rol_instance
        )
        return usuario

    def update(self, instance, validated_data):
        from .models import DetalleAlquiler, Inventario, Historialinventario
        from django.db import transaction

        materiales_data = validated_data.pop("materiales", [])
        hubo_danos = validated_data.pop("hubo_danos_perdida", False)
        danos_data = validated_data.pop("danos", [])

        with transaction.atomic():
            # actualizar cliente
            cliente = instance.cliente
            cliente.nombre_completo = validated_data["cliente_nombre"]
            cliente.telefono = validated_data["cliente_telefono"]
            cliente.save()

            # actualizar alquiler
            instance.fecha_evento = validated_data["fecha_evento"]
            instance.fecha_recojo = validated_data["fecha_recojo"]
            instance.direccion_entrega = validated_data.get("direccion_entrega", "")
            direccion_entrega = validated_data.get("direccion_entrega", "")

            # obtener usuario autenticado mapeado a Usuarios
            request = self.context.get("request")
            usuario_hist = None
            if request and hasattr(request, "user") and isinstance(request.user, Usuario):
                try:
                    usuario_hist = Usuarios.objects.get(usuario_id=request.user.usuario_id)
                except Usuarios.DoesNotExist:
                    usuario_hist = None

            # borrar historial previo asociado a detalles de este alquiler
            detalles_previos = list(DetalleAlquiler.objects.filter(alquiler=instance))
            if detalles_previos:
                Historialinventario.objects.filter(detalle__in=detalles_previos).delete()

            # borrar detalles anteriores
            DetalleAlquiler.objects.filter(alquiler=instance).delete()

            total_alquiler = Decimal("0.00")
            total_danos = Decimal("0.00")
            detalles_creados = {}

            # recrear materiales
            for m in materiales_data:
                inventario_id = m["inventario_id"]
                cantidad = m["cantidad"]

                inv = Inventario.objects.get(pk=inventario_id)
                precio_alq, precio_dan = self._obtener_precios(inv)

                subtotal = precio_alq * cantidad
                total_alquiler += subtotal

                detalle = DetalleAlquiler.objects.create(

                    inventario=inv,
                    cantidad_alquilada=cantidad,
                    direccion_entrega=direccion_entrega,

                    cantidad_danada=0,
                )

                detalles_creados[inventario_id] = detalle

            # registrar daños / pérdidas
            if hubo_danos:
                for d in danos_data:
                    inventario_id = d["inventario_id"]
                    cantidad = d["cantidad"]
                    tipo = d["tipo"]

                    inv = Inventario.objects.get(pk=inventario_id)
                    _, precio_dan = self._obtener_precios(inv)

                    subtotal_dano = precio_dan * cantidad
                    total_danos += subtotal_dano

                    detalle_rel = detalles_creados.get(inventario_id)

                    # si el material dañado no está en los materiales recreados, crear detalle base
                    if detalle_rel is None:
                        detalle_rel = DetalleAlquiler.objects.create(
                            alquiler=instance,
                            inventario=inv,
                            cantidad_alquilada=0,
                            direccion_entrega=direccion_entrega,

                            cantidad_devuelta_ok=0,
                            cantidad_danada=cantidad,
                        )
                        detalles_creados[inventario_id] = detalle_rel
                    else:
                        detalle_rel.cantidad_danada = (detalle_rel.cantidad_danada or 0) + cantidad
                        detalle_rel.save(update_fields=["cantidad_danada"])

                    Historialinventario.objects.create(
                        inventario=inv,
                        fecha=timezone.now(),
                        tipo_movimiento="Salida",
                        cantidad=cantidad,
                        motivo=f"{tipo} en el alquiler N° {instance.alquiler_id}",
                        usuario=usuario_hist,
                        detalle=detalle_rel,
                    )

            instance.total_orden_alquiler = total_alquiler
            instance.total_orden_danos = total_danos
            instance.save(update_fields=[
                "fecha_evento",
                "fecha_recojo",
                "total_orden_alquiler",
                "total_orden_danos",
            ])

class InventarioViewSet(viewsets.ModelViewSet):
    """
    API endpoint que muestra todo el inventario CON su stock actual calculado
    y permite crear / editar / borrar artículos.
    """
    serializer_class = InventarioSerializer
    permission_classes = [IsAuthenticated]
    queryset = Inventario.objects.all()

    def get_queryset(self):
        stock_calculado = Sum(
            Case(
                When(
                    historialinventario__tipo_movimiento='Entrada',
                    then=F('historialinventario__cantidad'),
                ),
                When(
                    historialinventario__tipo_movimiento='Salida',
                    then=F('historialinventario__cantidad') * -1,
                ),
                default=0,
                output_field=IntegerField(),
            )
        )
        queryset = (
            Inventario.objects
            .annotate(stock_actual=Coalesce(stock_calculado, 0))
            .order_by('nombre_articulo')
        )
        return queryset

    @transaction.atomic
    def destroy(self, request, *args, **kwargs):
        """
        BORRADO TOTAL (opción A):
        - Borra historial de movimientos
        - Borra lista de precios de alquiler
        - Borra lista de precios de daños
        - Borra el artículo de inventario
        """
        instance = self.get_object()

        # Primero borramos todo lo que depende de este inventario
        Historialinventario.objects.filter(inventario=instance).delete()
        Listapreciosalquiler.objects.filter(inventario=instance).delete()
        Listapreciosdanos.objects.filter(inventario=instance).delete()

        # Finalmente borramos el inventario
        return super().destroy(request, *args, **kwargs)

class ListaPreciosAlquilerView(APIView):
    permission_classes = [IsAuthenticated]

    def get(self, request):
        qs = (
            Listapreciosalquiler.objects
            .select_related('inventario')
            .order_by('inventario__nombre_articulo')
        )
        serializer = ListaPreciosAlquilerSerializer(qs, many=True)
        return Response(serializer.data)

    def put(self, request):
        """
        Actualiza la lista completa de precios de alquiler.
        """
        data = request.data

        # ids que vienen desde el frontend
        ids_enviados = [
            item.get("precio_alquiler_id")
            for item in data
            if item.get("precio_alquiler_id") is not None
        ]

        if ids_enviados:
            Listapreciosalquiler.objects.exclude(
                precio_alquiler_id__in=ids_enviados
            ).delete()
        else:
            Listapreciosalquiler.objects.all().delete()

        for item in data:
            pk = item.get("precio_alquiler_id")

            defaults = {
                "inventario_id": item["inventario_id"],
                "nombre_tarifa": item.get("nombre_tarifa") or "Precio por Unidad",
                "unidades_incluidas": item["unidades_incluidas"],
                "precio_tarifa": item["precio_tarifa"],
            }

            if pk:
                Listapreciosalquiler.objects.update_or_create(
                    precio_alquiler_id=pk,
                    defaults=defaults,
                )
            else:
                Listapreciosalquiler.objects.create(**defaults)

        qs = (

class ListaPreciosDanosView(APIView):
    permission_classes = [IsAuthenticated]

    def get(self, request):
        qs = (
            Listapreciosdanos.objects
            .select_related('inventario')
            .order_by('inventario__nombre_articulo')
        )
        serializer = ListaPreciosDanosSerializer(qs, many=True)
        return Response(serializer.data)

    def put(self, request):
        """
        Actualiza la lista completa de precios de daños.
        """
        data = request.data

        ids_enviados = [
            item.get("precio_dano_id")
            for item in data
            if item.get("precio_dano_id") is not None
        ]

        if ids_enviados:
            Listapreciosdanos.objects.exclude(
                precio_dano_id__in=ids_enviados
            ).delete()
        else:
            Listapreciosdanos.objects.all().delete()

        for item in data:
            pk = item.get("precio_dano_id")

            defaults = {
                "inventario_id": item["inventario_id"],
                "nombre_tarifa": item.get("nombre_tarifa") or item.get("detalle", ""),
                "unidades_incluidas": item["unidades_incluidas"],
                "precio_tarifa": item["precio_tarifa"],
            }

            if pk:
                Listapreciosdanos.objects.update_or_create(
                    precio_dano_id=pk,
                    defaults=defaults,
                )
            else:
                Listapreciosdanos.objects.create(**defaults)

        qs = (
            Listapreciosdanos.objects
            .select_related('inventario')
            .order_by('inventario__nombre_articulo')
        )
        serializer = ListaPreciosDanosSerializer(qs, many=True)
        return Response(serializer.data)

"""Gestion de usuarios"""
class UsuarioViewSet(viewsets.ModelViewSet):
    """
    Gestiona usuarios: Listar, Crear, Borrar.
    SOLO accesible si el usuario logueado es 'Administrador'.
    """
    permission_classes = [IsAuthenticated]
    
    # Por defecto usa el serializer de lectura
    serializer_class = UsuarioSerializer

    def get_queryset(self):
        # Listar todos los usuarios
        return Usuario.objects.all().select_related('rol').order_by('usuario_id')

    def get_serializer_class(self):
        if self.action == 'create':
            return UsuarioCreateSerializer
        # Agregamos esta condición para usar el serializador de edición
        if self.action in ['update', 'partial_update']:
            return UsuarioUpdateSerializer
        return UsuarioSerializer

    def perform_create(self, serializer):
        # Validar permisos: Solo Administrador puede crear
        if self.request.user.rol.nombre_rol != 'Administrador':
            raise PermissionDenied("No tienes permisos para crear usuarios.")
        serializer.save()

    def perform_destroy(self, instance):
        # Validar permisos: Solo Administrador puede borrar
        if self.request.user.rol.nombre_rol != 'Administrador':
            raise PermissionDenied("No tienes permisos para borrar usuarios.")
        # Evitar que se borre a sí mismo (opcional pero recomendado)
        if instance.usuario_id == self.request.user.usuario_id:
             raise PermissionDenied("No puedes eliminar tu propio usuario.")
        instance.delete()

class AlquilerListCreateView(APIView):
    def get(self, request):
        # traemos alquileres y su cliente
        alquileres = Alquileres.objects.all().order_by('-fecha_registro')

        data = []
        for a in alquileres:
            # traer los detalles de este alquiler

                .filter(alquiler_id=a.alquiler_id)
                .select_related('inventario')
            )

            detalles = []
            direccion_entrega = ""

            for d in detalles_qs:
                inv = d.inventario
                # armamos un nombre legible: "Mesa - blanca" por ejemplo
                nombre_item = getattr(inv, "nombre_articulo", str(inv))
                detalle_extra = getattr(inv, "detalle", None)
                if detalle_extra:
                    nombre_item = f"{nombre_item} - {detalle_extra}"

                # tomamos la dirección del primer detalle que la tenga
                if not direccion_entrega and d.direccion_entrega:
                    direccion_entrega = d.direccion_entrega

                detalles.append({
                    "cantidad_alquilada": d.cantidad_alquilada,
                    "inventario_id": getattr(inv, "inventario_id", getattr(inv, "id", None)),
                    "inventario_nombre": nombre_item,
                    "precio_alquiler_snapshot": d.precio_alquiler_snapshot,
                    "subtotal_alquiler": d.subtotal_alquiler,
                    "direccion_entrega": d.direccion_entrega,
                })

            data.append({
                "alquiler_id": a.alquiler_id,
                "cliente_nombre": getattr(a.cliente, "nombre_completo", ""),
                "cliente_telefono": getattr(a.cliente, "telefono", ""),

                "direccion_entrega": direccion_entrega,
                "detalles": detalles,  # materiales
            })

        return Response(data, status=status.HTTP_200_OK)

  tipo_movimiento: 'Salida',
  cantidad: 1,
  motivo: '',
});

// Estado del componente
const inventarioList = ref([]);
const loading = ref(false);
const loadingItems = ref(false);
const error = ref(null);

// Cargar artículos para el dropdown
onMounted(async () => {
  loadingItems.value = true;
  try {
    const response = await getInventario();
    inventarioList.value = response.data;
  } catch (err) {
    console.error('Error al cargar inventario en modal:', err);
    error.value = 'Error al cargar la lista de artículos.';
  } finally {
    loadingItems.value = false;
  }
});

const cerrar = () => {
  emit('cerrar');
};

const guardar = async () => {
  if (loading.value) return;

  loading.value = true;
  error.value = null;

  // Validaciones
  if (!movimiento.value.inventario) {
    error.value = 'Por favor, seleccione un artículo.';
    loading.value = false;
    return;
  }
  if (!movimiento.value.cantidad || movimiento.value.cantidad <= 0) {
    error.value = 'La cantidad debe ser mayor a 0.';
    loading.value = false;
    return;
  }
  if (!movimiento.value.motivo.trim()) {
    error.value = 'Por favor, ingrese un motivo.';
    loading.value = false;
    return;
  }

  try {
    await createHistorialMovimiento({
      inventario: movimiento.value.inventario,
      tipo_movimiento: movimiento.value.tipo_movimiento,
      cantidad: movimiento.value.cantidad,
      motivo: movimiento.value.motivo,
    });

    emit('guardar');
  } catch (err) {
    console.error('Error en guardar():', err);

    if (err.response && err.response.data) {
      if (typeof err.response.data === 'object') {
        const errors = Object.entries(err.response.data).map(([key, value]) => {
          return `${key}: ${Array.isArray(value) ? value.join(', ') : value}`;
        });
        error.value = `Error de validación: ${errors.join('. ')}`;
      } else {
        error.value =
          err.response.data.detail || 'Error al guardar. Revise los campos.';
      }
    } else {
      error.value = 'Error al guardar el movimiento. Intente de nuevo.';
    }
  } finally {
    loading.value = false;
  }
};

<script setup>

const guardando = ref(false);

const esCotizacion = computed(() => guardarComoCotizacion.value);

const textoBotonGuardar = computed(() => {
  if (guardando.value) {
    return esCotizacion.value ? 'Guardando cotización...' : 'Guardando alquiler...';
  }
  return esCotizacion.value ? 'Guardar Cotización' : 'Guardar Alquiler';
});
// para que no falle el :vista en InicioPage.vue
const props = defineProps({
  vista: {
    type: String,
    default: 'pendientes',
  },
});

const clienteNombre = ref('');
const clienteTelefono = ref('');

const fechaEvento = ref('');
const fechaRecojo = ref('');

const usaGarantia = ref(false);
const garantiaTipo = ref('');
const garantiaValor = ref(0);
const garantiaDescripcion = ref('');
const direccionEntrega = ref('');
const guardarComoCotizacion = ref(false);
// inventario y precios
const inventarioLista = ref([]);
const preciosMapa = ref({}); // { inventario_id: precio }

const loading = ref(false);
const error = ref(null);

// detalle de materiales en el formulario
const materiales = ref([crearFilaMaterial()]);

function crearFilaMaterial() {
  return {
    cantidad: 1,
    nombreBusqueda: '',
    inventarioId: null,
    precio: 0,
    subtotal: 0,
    resultados: [],
  };
}

// Cargar inventario y lista de precios cuando se abre el modal
onMounted(async () => {
  loading.value = true;
  error.value = null;
  try {
    const [invRes, preciosRes] = await Promise.all([
      getInventario(),             // GET /inventario/
      fetchListaPreciosAlquiler(), // GET /lista-precios/alquiler/
    ]);

    inventarioLista.value = invRes.data || [];

    const mapa = {};
    (preciosRes.data || []).forEach((p) => {
      // viene como inventario_id y precio_tarifa
      mapa[p.inventario_id] = Number(p.precio_tarifa);
    });
    preciosMapa.value = mapa;
  } catch (err) {
    console.error('Error cargando inventario o precios', err);
    error.value = 'No se pudo cargar inventario o precios.';
  } finally {
    loading.value = false;
  }
});

// --------- BUSCADOR DE MATERIALES ---------
const buscarMaterial = (index) => {
  const term = materiales.value[index].nombreBusqueda.trim().toLowerCase();
  if (!term) {
    materiales.value[index].resultados = [];
    return;
  }

  const resultados = inventarioLista.value
    .filter((item) => {
      const nombre = (item.nombre_articulo || '').toLowerCase();
      const detalle = (item.detalle || '').toLowerCase();
      return nombre.includes(term) || detalle.includes(term);
    })
    .slice(0, 10)
    .map((item) => ({
      ...item,
      // OJO: aquí volvemos a usar item.id
      precio: preciosMapa.value[item.id] ?? null,
    }));

  materiales.value[index].resultados = resultados;
};

const seleccionarMaterial = (index, material) => {
  const fila = materiales.value[index];

  // el ID que vamos a mandar al backend
  fila.inventarioId = material.id;

  fila.nombreBusqueda =
    material.nombre_articulo +
    (material.detalle ? ' - ' + material.detalle : '');

  // el precio lo sacamos con ese mismo id
  fila.precio = preciosMapa.value[material.id] ?? 0;
  fila.resultados = [];

  actualizarSubtotal(index);
};



// --------- SUBTOTALES Y TOTAL ---------
const actualizarSubtotal = (index) => {
  const fila = materiales.value[index];
  const cantidad = Number(fila.cantidad) || 0;
  const precio = Number(fila.precio) || 0;
  fila.subtotal = cantidad * precio;
};

const totalFinal = computed(() =>
  materiales.value.reduce(
    (acc, fila) => acc + (Number(fila.subtotal) || 0),
    0
  )
);

// --------- GESTIÓN DE FILAS ---------
const agregarMaterial = () => {
  materiales.value.push(crearFilaMaterial());
};

const quitarMaterial = (index) => {
  materiales.value.splice(index, 1);
  if (materiales.value.length === 0) {
    materiales.value.push(crearFilaMaterial());
  }
};

// --------- GUARDAR (EMITIR PAYLOAD AL PADRE) ---------
 const guardar = () => {
  if (guardando.value) return;

  if (!clienteNombre.value.trim()) {
    alert('Por favor, ingresa el nombre del cliente.');
    return;
  }
  if (!clienteTelefono.value.trim()) {
    alert('Por favor, ingresa el teléfono.');
    return;
  }
  if (!fechaEvento.value) {
    alert('Por favor, selecciona la fecha del evento.');
    return;
  }
  if (!fechaRecojo.value) {
    alert('Por favor, selecciona la fecha de recojo.');
    return;
  }

  const materialesValidos = materiales.value.filter(
    (m) => m.inventarioId && m.cantidad > 0
  );

  if (!materialesValidos.length) {
    alert('Agrega al menos un material al alquiler.');
    return;
  }

  if (!direccionEntrega.value.trim()) {
    alert('Por favor, ingresa la dirección de entrega.');
    return;
  }

  const payload = {
    cliente_nombre: clienteNombre.value.trim(),
    cliente_telefono: clienteTelefono.value.trim(),
    fecha_evento: fechaEvento.value,
    fecha_recojo: fechaRecojo.value,
    direccion_entrega: direccionEntrega.value.trim(),
    estado: guardarComoCotizacion.value ? 'Cotización' : 'Reservado',
    materiales: materialesValidos.map((m) => ({
      inventario_id: m.inventarioId,
      cantidad: m.cantidad,
    })),
  };

  if (usaGarantia.value) {
    payload.garantia_tipo = garantiaTipo.value;
    payload.garantia_descripcion = garantiaDescripcion.value || '';
    if (garantiaTipo.value === 'Dinero') {
      payload.garantia_valor = garantiaValor.value || 0;
    }
  }

  guardando.value = true;

};

<script setup>
import { ref, onMounted, computed } from 'vue';
import { getInventario } from '@/services/inventarioService.js';
import {
  fetchListaPreciosAlquiler,
  fetchListaPreciosDanos,
} from '@/services/prices.js';

const props = defineProps({
  pedido: { type: Object, required: true },
  vista: { type: String, required: true },
});

const emit = defineEmits([
  'cerrar',
  'cancelar-alquiler',
  'guardar',
  'entregar',
  'recoger',
  'confirmar-alquiler',
]);

const enModoEdicion = ref(false);
const pedidoEditable = ref(null);

// inventario y precios para los buscadores
const inventarioLista = ref([]);
const preciosMapa = ref({}); // { inventario_id: precio_alquiler }
const preciosDanosMapa = ref({}); // { inventario_id: precio_dano }

const loadingInv = ref(false);
const errorInv = ref(null);

const mostrarSeccionDanos = computed(() => {
  if (!pedidoEditable.value) return false;
  return props.vista === 'porRecoger' || props.pedido.estado === 'Entregado';
});

onMounted(async () => {
  loadingInv.value = true;
  errorInv.value = null;

  try {
    const [invRes, preciosRes, danosRes] = await Promise.all([
      getInventario(),
      fetchListaPreciosAlquiler(),
      fetchListaPreciosDanos(),
    ]);

    inventarioLista.value = invRes.data || [];

    const mapaAlquiler = {};
    (preciosRes.data || []).forEach((p) => {
      mapaAlquiler[p.inventario_id] = Number(p.precio_tarifa);
    });
    preciosMapa.value = mapaAlquiler;

    const mapaDanos = {};
    (danosRes.data || []).forEach((p) => {
      mapaDanos[p.inventario_id] = Number(p.precio_tarifa);
    });
    preciosDanosMapa.value = mapaDanos;
  } catch (err) {
    console.error('Error cargando inventario o precios', err);
    errorInv.value = 'No se pudo cargar inventario o precios.';
  } finally {
    loadingInv.value = false;
  }
});

const onOverlayClick = () => {
  if (!enModoEdicion.value) {
    emit('cerrar');
  }
};

const formatearMonto = (valor) => {
  const numero = Number(valor || 0);
  return Number.isInteger(numero) ? numero : numero.toFixed(2);
};

const prepararDetalleEditable = (detalle) => {
  return (detalle || []).map((item) => ({
    ...item,
    inventarioId: item.inventarioId ?? null,
    nombreBusqueda: item.item || '',
    resultados: [],
    cant: Number(item.cant) || 1,
    precioUnitario: Number(item.precioUnitario) || 0,
    total: Number(item.total) || 0,
  }));
};

const prepararDanosEditable = (danosDetalle) => {
  return (danosDetalle || []).map((item) => ({
    inventarioId: item.inventarioId ?? null,
    item: item.item || '',
    nombreBusqueda: item.item || '',
    tipo: item.tipo || 'Daño',
    cantidad: Number(item.cantidad) || 1,
    precioUnitario: Number(item.precioUnitario) || 0,
    total: Number(item.total) || 0,
    resultados: [],
  }));
};

const recalcularTotalGeneral = () => {
  if (!pedidoEditable.value) return;
  pedidoEditable.value.totalGeneral =
    Number(pedidoEditable.value.granTotal || 0) +
    Number(pedidoEditable.value.totalDanos || 0);
};

const entrarModoEdicion = () => {
  const copia = JSON.parse(JSON.stringify(props.pedido));

  copia.detalle = prepararDetalleEditable(copia.detalle);
  copia.danosDetalle = prepararDanosEditable(copia.danosDetalle);

  if (copia.garantia == null) {
    copia.garantia = false;
  }

  copia.huboDanosPerdida = Array.isArray(copia.danosDetalle) && copia.danosDetalle.length > 0;
  copia.totalDanos = Number(copia.totalDanos || 0);
  copia.granTotal = Number(copia.granTotal || 0);
  copia.totalGeneral = Number(copia.granTotal || 0) + Number(copia.totalDanos || 0);

  pedidoEditable.value = copia;
  enModoEdicion.value = true;

  actualizarTotales();
  actualizarTotalesDanos();
};

const cancelarEdicion = () => {
  enModoEdicion.value = false;
  pedidoEditable.value = null;
};

const guardarCambios = () => {
  if (!pedidoEditable.value) return;

  pedidoEditable.value.ubicacion = pedidoEditable.value.direccion;

  // limpieza de materiales
  pedidoEditable.value.detalle = (pedidoEditable.value.detalle || [])
    .filter((item) => item.inventarioId && Number(item.cant) > 0)
    .map((item) => ({
      ...item,
      cant: Number(item.cant) || 0,
      precioUnitario: Number(item.precioUnitario) || 0,
      total: Number(item.total) || 0,
      item: item.item || item.nombreBusqueda || '',
    }));

  // limpieza de daños
  if (!pedidoEditable.value.huboDanosPerdida) {
    pedidoEditable.value.danosDetalle = [];
    pedidoEditable.value.totalDanos = 0;
  } else {
    pedidoEditable.value.danosDetalle = (pedidoEditable.value.danosDetalle || [])
      .filter((item) => item.inventarioId && Number(item.cantidad) > 0)
      .map((item) => ({
        ...item,
        cantidad: Number(item.cantidad) || 0,
        precioUnitario: Number(item.precioUnitario) || 0,
        total: Number(item.total) || 0,
        tipo: item.tipo || 'Daño',
        item: item.item || item.nombreBusqueda || '',
      }));
  }

  recalcularTotalGeneral();

  emit('guardar', pedidoEditable.value);
  enModoEdicion.value = false;
};

const actualizarTotales = () => {
  if (!pedidoEditable.value || !Array.isArray(pedidoEditable.value.detalle)) {
    return;
  }

  let nuevoGranTotal = 0;

  pedidoEditable.value.detalle.forEach((item) => {
    const cant = Number(item.cant) || 0;
    const precio = Number(item.precioUnitario) || 0;
    item.total = cant * precio;
    nuevoGranTotal += item.total;
  });

  pedidoEditable.value.granTotal = nuevoGranTotal;
  recalcularTotalGeneral();
};

const actualizarTotalesDanos = () => {
  if (!pedidoEditable.value || !Array.isArray(pedidoEditable.value.danosDetalle)) {
    return;
  }

  let totalDanos = 0;

  pedidoEditable.value.danosDetalle.forEach((item) => {
    const cantidad = Number(item.cantidad) || 0;
    const precio = Number(item.precioUnitario) || 0;
    item.total = cantidad * precio;
    totalDanos += item.total;
  });

  pedidoEditable.value.totalDanos = totalDanos;
  recalcularTotalGeneral();
};

const agregarMaterial = () => {
  if (!pedidoEditable.value) return;

  if (!Array.isArray(pedidoEditable.value.detalle)) {
    pedidoEditable.value.detalle = [];
  }

  pedidoEditable.value.detalle.push({
    inventarioId: null,
    cant: 1,
    item: '',
    nombreBusqueda: '',
    precioUnitario: 0,
    total: 0,
    resultados: [],
  });
};

const quitarMaterial = (index) => {
  if (!pedidoEditable.value || !Array.isArray(pedidoEditable.value.detalle)) return;
  if (pedidoEditable.value.detalle.length === 1) return;

  pedidoEditable.value.detalle.splice(index, 1);
  actualizarTotales();
};

const crearFilaDano = () => ({

  item: '',
  nombreBusqueda: '',
  tipo: 'Daño',
  cantidad: 1,
  precioUnitario: 0,
  total: 0,
  resultados: [],
});

const onCambioHuboDanos = () => {
  if (!pedidoEditable.value) return;

  if (pedidoEditable.value.huboDanosPerdida) {
    if (!Array.isArray(pedidoEditable.value.danosDetalle)) {
      pedidoEditable.value.danosDetalle = [];
    }
    if (pedidoEditable.value.danosDetalle.length === 0) {
      pedidoEditable.value.danosDetalle.push(crearFilaDano());
    }
  } else {
    pedidoEditable.value.danosDetalle = [];
    pedidoEditable.value.totalDanos = 0;
    recalcularTotalGeneral();
  }

const agregarDano = () => {
  if (!pedidoEditable.value) return;

  if (!Array.isArray(pedidoEditable.value.danosDetalle)) {
    pedidoEditable.value.danosDetalle = [];
  }

  pedidoEditable.value.danosDetalle.push(crearFilaDano());
};

const quitarDano = (index) => {
  if (!pedidoEditable.value || !Array.isArray(pedidoEditable.value.danosDetalle)) return;

  pedidoEditable.value.danosDetalle.splice(index, 1);

  if (pedidoEditable.value.danosDetalle.length === 0) {
    pedidoEditable.value.huboDanosPerdida = false;
  }

  actualizarTotalesDanos();
};

// BUSCADOR DE MATERIALES
const buscarMaterial = (index) => {
  if (!pedidoEditable.value) return;

  }

    })
    .slice(0, 10)
    .map((item) => ({

    }));

const seleccionarMaterial = (index, material) => {

  return `${parts[2]}/${parts[1]}/${parts[0]}`;
};

<script setup>
import { ref, onMounted, computed } from 'vue';
import { fetchHistorialAlquileres } from '@/services/alquileres';
import { usuarioActual } from '@/services/auth.js';
// Corregimos el nombre del import para mayor claridad
import RevisarAlquilerModal from '@/components/RevisarAlquilerModal.vue';

const modalVisible = ref(false);
const alquilerSeleccionado = ref(null);
const historiales = ref([]);
const loading = ref(true);

// Variables de Filtro
const busqueda = ref('');
const mostrarFiltros = ref(false);
const filtroEstado = ref('');
const fechaInicio = ref('');
const fechaFin = ref('');

onMounted(async () => {
  try {
    const response = await fetchHistorialAlquileres();
    historiales.value = response.data;
  } catch (error) {
    console.error("Error cargando el historial:", error);
  } finally {
    loading.value = false;
  }
});

// LÓGICA DE FILTRADO AVANZADO
const historialesFiltrados = computed(() => {
  return historiales.value.filter(h => {
    // 1. Filtro de Texto (Cliente, ID o Teléfono)
    const texto = busqueda.value.toLowerCase();
    const matchTexto = !texto || 
      (h.nombre_cliente && h.nombre_cliente.toLowerCase().includes(texto)) ||
      String(h.alquiler_id).includes(texto) ||
      (h.telefono_cliente && h.telefono_cliente.includes(texto));

    // 2. Filtro de Estado
    const matchEstado = !filtroEstado.value || h.estado === filtroEstado.value;

    // 3. Filtro de Rango de Fechas (basado en fecha_evento)
    let matchFecha = true;
    if (fechaInicio.value) {
      matchFecha = matchFecha && (h.fecha_evento >= fechaInicio.value);
    }
    if (fechaFin.value) {
      matchFecha = matchFecha && (h.fecha_evento <= fechaFin.value);
    }

    return matchTexto && matchEstado && matchFecha;
  });
});

const formatDate = (dateString) => {
  if (!dateString) return '';
  const parts = dateString.split('-');
  return `${parts[2]}/${parts[1]}/${parts[0]}`;
};

// Función para obtener clase de color según estado
const getEstadoClass = (estado) => {
  if (estado === 'Completado') return 'text-success'; // Verde
  if (estado === 'Cancelado') return 'text-danger';   // Rojo
  return '';
};

const limpiarFiltros = () => {
  filtroEstado.value = '';
  fechaInicio.value = '';
  fechaFin.value = '';
  busqueda.value = '';
};

const abrirModal = (alquiler) => {
  alquilerSeleccionado.value = alquiler;
  modalVisible.value = true;
};

const cerrarModal = () => {
  modalVisible.value = false;
  alquilerSeleccionado.value = null;
};

<script setup>
import { ref, watch, computed } from 'vue';
import { useRoute, useRouter } from 'vue-router';

  const itemName = route.query.item_nombre;
  return itemName ? `Historial: ${itemName}` : 'Historial de Inventario';
});

// Carga de datos
const cargarHistorial = async (query) => {
  try {
    loading.value = true;
    error.value = null;

    const filtros = {};
    if (query.item_id) {
      filtros.inventario = query.item_id;
    }

    console.log('DEBUG route.query en HistorialInventarioPage:', query);

    const response = await getHistorialInventario(filtros);
    historial.value = response.data;
  } catch (err) {
    console.error(err);
    error.value = 'Error al cargar el historial. Intente de nuevo.';
  } finally {
    loading.value = false;
  }
};

// Filtro de búsqueda
const filteredHistorial = computed(() => {
  if (!searchTerm.value) {
    return historial.value;
  }
  const lowerSearch = searchTerm.value.toLowerCase();

  return historial.value.filter((log) => {
    const inMotivo =
      log.motivo && log.motivo.toLowerCase().includes(lowerSearch);
    const inArticulo =
      log.nombre_articulo &&
      log.nombre_articulo.toLowerCase().includes(lowerSearch);
    const inUsuario =
      log.nombre_usuario &&
      log.nombre_usuario.toLowerCase().includes(lowerSearch);

    return inMotivo || inArticulo || inUsuario;
  });
});

// Reaccionar a cambios en la URL
watch(
  () => route.query,
  (newQuery) => {
    cargarHistorial(newQuery);
  },

const mapAlquilerToCard = (a) => {
  const estadoNormalizado = normalizarTexto(a.estado);

  const detalle = (a.detalles || []).map((d) => ({
    inventarioId: d.inventario_id,
    cant: Number(d.cantidad_alquilada || 0),
    item: d.inventario_nombre,
    precioUnitario: Number(d.precio_alquiler_snapshot || 0),
    total: Number(d.subtotal_alquiler || 0),
  }));

  const danosDetalle = (a.detalles || [])
    .filter((d) => Number(d.cantidad_danada || 0) > 0)

      tipo: 'Daño',
      precioUnitario: Number(d.precio_danos_snapshot || 0),
      total:

        Number(d.precio_danos_snapshot || 0),
    }));

  const totalAlquiler = Number(a.total_orden_alquiler || 0);
  const totalDanos = Number(a.total_orden_danos || 0);

  return {
    numeroPedido: a.alquiler_id,
    cliente: a.cliente_nombre || 'Sin nombre',
    telefono: a.cliente_telefono || '',
    ubicacion: a.direccion_entrega || '',

    loadingPedidos.value = false;

    await cargarInventario();

    mostrarToast('Artículo actualizado correctamente.', 'success');
  } catch (err) {
    console.error(err);
    mostrarToast('Error al actualizar el artículo.', 'error');
  } finally {
    saving.value = false;
  }
};


const borrarItem = async (item) => {
  if (!confirm(`¿Seguro que quieres borrar "${item.nombre_articulo}"?`)) {
    return;
  }

  try {
    await deleteInventarioItem(item.id);
    await cargarInventario();
    mostrarToast('Artículo eliminado correctamente.', 'success');
  } catch (err) {
    console.error(err);
    const detail = err.response?.data?.detail;
    mostrarToast(detail || 'No se pudo eliminar el artículo.', 'error');
  }
};

<script setup>
import { ref, computed, onMounted } from 'vue';
import { usuarioActual } from '@/services/auth';
import EditarPreciosModal from '@/components/EditarPreciosModal.vue';

import {
  fetchListaPreciosAlquiler,
  fetchListaPreciosDanos,
  saveListaPreciosAlquiler,
  saveListaPreciosDanos,
} from '@/services/prices';

/* Estado general */
const modalVisible = ref(false);
const vistaActiva = ref('precios');
const guardando = ref(false);
const loading = ref(true);

/* Listas principales */
const listaPrecios = ref([]);
const listaPreciosDanos = ref([]);

/* Toast */
const toastVisible = ref(false);
const toastMessage = ref('');
const toastType = ref('success');
let toastTimeoutId = null;

const mostrarToast = (message, type = 'success') => {
  toastMessage.value = message;
  toastType.value = type;
  toastVisible.value = true;

  if (toastTimeoutId) {
    clearTimeout(toastTimeoutId);
  }

  toastTimeoutId = setTimeout(() => {
    toastVisible.value = false;
  }, 3000);
};

/* Cargar datos desde API */
const cargarDatos = async () => {
  loading.value = true;
  try {
    const [alq, dan] = await Promise.all([
      fetchListaPreciosAlquiler(),
      fetchListaPreciosDanos(),
    ]);

    // Lista de precios de ALQUILER
    listaPrecios.value = alq.data.map(i => ({
      id: i.precio_alquiler_id,
      cantidad: i.unidades_incluidas,
      detalle: i.nombre_articulo || i.nombre_tarifa,
      precioUnitario: Number(i.precio_tarifa),
      total: Number(i.unidades_incluidas) * Number(i.precio_tarifa),
      inventario_id: i.inventario_id,
      nombre_tarifa: i.nombre_tarifa,
    }));

    // Lista de precios de DAÑOS
    listaPreciosDanos.value = dan.data.map(i => {
      const unidades = i.unidades_incluidas ?? 1;
      return {
        id: i.precio_dano_id,
        cantidad: unidades,
        detalle: i.nombre_articulo || i.nombre_tarifa,
        precioUnitario: Number(i.precio_tarifa),
        total: unidades * Number(i.precio_tarifa),
        inventario_id: i.inventario_id,
        nombre_tarifa: i.nombre_tarifa,
      };
    });

  } catch (err) {
    console.error(err);
    mostrarToast('Error al cargar las listas de precios.', 'error');
  }
  loading.value = false;
};

onMounted(cargarDatos);

/* Lógica de tabla activa */
const datosTablaActiva = computed(() =>
  vistaActiva.value === 'precios'
    ? listaPrecios.value
    : listaPreciosDanos.value
);

/* Modal */
const abrirModal = () => { modalVisible.value = true; };
const cerrarModal = () => { modalVisible.value = false; };

/* Guardar cambios */
const guardarCambios = async (listaEditada) => {
  guardando.value = true;

  try {
    if (vistaActiva.value === 'precios') {
      const payload = listaEditada.map(i => ({
        precio_alquiler_id: i.id,
        inventario_id: i.inventario_id,
        nombre_tarifa: i.nombre_tarifa || 'Precio por Unidad',
        unidades_incluidas: i.cantidad,
        precio_tarifa: i.precioUnitario,
      }));
      await saveListaPreciosAlquiler(payload);
      listaPrecios.value = listaEditada;

      mostrarToast('Precios de alquiler guardados correctamente.');

    } else {
      const payload = listaEditada.map(i => ({
        precio_dano_id: i.id,
        inventario_id: i.inventario_id,
        nombre_tarifa: i.nombre_tarifa || 'Daño o pérdida',
        unidades_incluidas: i.cantidad,
        precio_tarifa: i.precioUnitario,
      }));
      await saveListaPreciosDanos(payload);
      listaPreciosDanos.value = listaEditada;

      mostrarToast('Precios de daños guardados correctamente.');
    }

    cerrarModal();

  } catch (err) {
    console.error(err);
    if (vistaActiva.value === 'precios') {
      mostrarToast('Error al guardar precios de alquiler.', 'error');
    } else {
      mostrarToast('Error al guardar precios de daños.', 'error');
    }
  }

  guardando.value = false;
};

import { createRouter, createWebHistory } from "vue-router";
import DashboardLayout from "@/components/layout/DashboardLayout.vue";
import { isAuthenticated, me, getUser, userRole } from "@/services/auth";

const router = createRouter({
  history: createWebHistory(),
  scrollBehavior() { return { top: 0 }; },
  routes: [
    { path: "/", redirect: { name: "login" } },

    // Público (no autenticado)
    {
      path: "/login",
      name: "login",
      component: () => import("@/pages/LoginView.vue"),
      meta: { guestOnly: true },
    },

    // Autenticado (layout privado)
    {
      path: "/app",
      component: DashboardLayout,
      meta: { requiresAuth: true },
      children: [
        { path: "", redirect: { name: "inicio" } },
        { path: "inicio", name: "inicio", component: () => import("@/pages/InicioPage.vue") },

        // Abiertas para ambos roles (ejemplo)
        { path: "historial-pedidos",   name: "historial-pedidos",   component: () => import("@/pages/HistorialAquileresPage.vue") },
        { path: "inventario",          name: "inventario",          component: () => import("@/pages/InventarioPage.vue") },
        { path: "lista-precios",       name: "lista-precios",       component: () => import("@/pages/ListaPreciosPage.vue") },
        { path: "historial-inventario",name: "historial-inventario",component: () => import("@/pages/HistorialInventarioPage.vue") },

        // SOLO Administrador
        {
          path: "usuarios",
          name: "usuarios",
          component: () => import("@/pages/UsuariosPage.vue"),
          meta: { roles: ["Administrador"] },
        },

        // ejemplo de ruta SOLO Propietaria (si la tuvieras):
        // { path: "reportes-propietaria", name:"reportes-propietaria", component: () => import("@/pages/ReportesPropietaria.vue"), meta: { roles: ["Propietaria"] } },
      ],
    },

    // 404 → login por ahora
    { path: "/:pathMatch(.*)*", redirect: { name: "login" } },
  ],
});

// -------- Guard global --------
let meAttempted = false;

router.beforeEach(async (to, from, next) => {
  const needsAuth = !!to.meta?.requiresAuth;
  const guestOnly = !!to.meta?.guestOnly;

  // 1) Si ya está autenticado y va a /login, redirige a /app
  if (guestOnly && isAuthenticated()) {
    return next({ name: "inicio" });
  }

  // 2) Si requiere auth y no hay token → login
  if (needsAuth && !isAuthenticated()) {
    return next({ name: "login" });
  }

  // 3) Si requiere auth y hay token pero no tenemos usuario en memoria,
  //    intenta recuperar desde /me una sola vez (para rellenar rol)
  if (needsAuth && isAuthenticated() && !getUser() && !meAttempted) {
    meAttempted = true;
    try {
      await me(); // si falla, caerá al catch
    } catch {
      return next({ name: "login" });
    }
  }

  // 4) Chequeo de roles por ruta (si la ruta define meta.roles)
  const allowed = to.meta?.roles;
  if (needsAuth && allowed?.length) {
    const role = userRole(); // "Administrador" | "Propietaria" | null
    if (!role || !allowed.includes(role)) {
      // opcional: muestra notificación de "No autorizado"
      return next({ name: "inicio" });
    }
  }

  next();
});

export default router;

import { ref } from "vue";
import { api, BASE } from "@/axios";
import axios from "axios";

// Estado reactivo del usuario en memoria
export const usuarioActual = ref(null);

// ---------- helpers de sesión ----------
const KEY_TOKEN = "token";
const KEY_USER  = "user";

// --- setSession (MODIFICADO) ---
// Ahora puede guardar token y usuario por separado sin borrar el otro
export function setSession({ token, user }) {
  if (token) {
    localStorage.setItem(KEY_TOKEN, token);
  }
  if (user) {
    localStorage.setItem(KEY_USER, JSON.stringify(user));
  }
  // Siempre lee desde localStorage para actualizar el estado
  usuarioActual.value = getUser(); 
}

export function clearSession() {
  localStorage.removeItem(KEY_TOKEN);
  localStorage.removeItem(KEY_USER);
  usuarioActual.value = null;
}

export function getToken() {
  return localStorage.getItem(KEY_TOKEN);
}

export function getUser() {
  const raw = localStorage.getItem(KEY_USER);
  try { return raw ? JSON.parse(raw) : null; } catch { return null; }
}

export function isAuthenticated() {
  return !!getToken();
}

export function userRole() {
  const u = usuarioActual.value || getUser();
  return u?.rol?.nombre_rol || null;
}

export function initAuthFromStorage() {
  const u = getUser();
  usuarioActual.value = u;
}

// ---------- llamadas al backend (CORREGIDAS) ----------

// --- login (Totalmente Modificado) ---
export async function login({ username, password }) {
  try {
    // Paso 1: Obtener el token de la nueva ruta de simplejwt
    // Usamos 'axios' base (no 'api') porque 'api' adjuntaría un token que no tenemos
    const res = await axios.post(`${BASE}/api/auth/token/`, {
      username,
      password,
    });

    const token = res.data.access; // simplejwt devuelve el token en 'access'
    if (!token) throw new Error("No se recibió token");

    // Paso 2: Guardar SOLO el token en sesión
    setSession({ token });

    // Paso 3: Llamar a 'me()' PARA OBTENER LOS DATOS DEL USUARIO
    // 'me()' usará la instancia 'api' que ahora adjuntará el nuevo token
    const user = await me(); // me() se encarga de guardar el usuario en sesión

    // Devolvemos ambos, tal como esperaba tu lógica anterior
    return { token, user };

  } catch (err) {
    clearSession(); // Limpia todo si falla
    // Muestra el error específico de "Credenciales inválidas"
    const errorMsg = err.response?.data?.detail || "Credenciales inválidas o error de red";
    throw new Error(errorMsg);
  }
}

// --- me (Corregido el Bug de la URL) ---
export async function me() {
  try {
    // BUG CORREGIDO:
    // Antes: api.get("/api/auth/me/") -> http://.../api/api/auth/me/ (MAL)
    // Ahora: api.get("/auth/me/")     -> http://.../api/auth/me/ (BIEN)
    const { data } = await api.get("/auth/me/");
    
    // Guardamos los datos del usuario en la sesión
    setSession({ user: data });
    return data;
  } catch (e) {
    clearSession(); // Si 'me' falla (ej. token expirado), deslogueamos
    throw e;
  }
}

export function logout() {
  clearSession();
}
